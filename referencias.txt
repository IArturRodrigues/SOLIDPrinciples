!!!!!!!!!!!!!! SEMPRE QUE CLONAR ESSE PROJETO (setup para API em Node) DELETE A PASTA .GIT !!!!!!!!!!!!!!
rm -rf .git/
em seguida der um git init

versoes {
   node: 16.15.1
   npm: 8.11.0
   yarn : 1.22.19
}

comandos {
   inicializar o package.json: (npm ou yarn) init -y (responder todas as perguntas com sim: -y)
   inicializar tsconfig.json: (npm ou yarn) tsc --init (depois de ter instalado o typescript)
   compilar typescript: yarn tsc

   opcionais {
      add eslint configs: yarn eslint --init
      add jest configs: yarn jest --init
   }
}

libs {
   essenciais {
      yarn add ts-node-dev -D
   }

   opcionais {
      yarn add eslint -D
      yarn add jest ts-jest -D
      yarn add -D @babel/cli @babel/node @babel/core @babel/preset-env @babel/preset-typescript babel-plugin-module-resolver (esse ultimo caso esteja utilizando o paths do tsconfig)
   }
}

possiveis erros {
   usando jest {
      setar ts-jest no preset do jest
      setar jest no eslint na part env "jest": true
      usando typescript paths, va no arquivo jest.config.js {
         const { compilerOptions } = require('./tsconfig.json');
         const { pathsToModuleNameMapper } = require('ts-jest'); // ou ts-jest/utils

         module.export = {
            moduleNameMapper: pathsToModuleNameMapper(compilerOptions.paths, { prefix: '<rootDir>' }),
         }
      }, se continuar dando erro va nas configuracoes do typescript e tire o .ts de includes
   }
      
   caso utilizando paths sem Babel: baixar tsc paths para que ele compile o codigo pra js sem erro
   caso utilizando paths com Babel: atualizar o babel config adicionando o novo path 
}
